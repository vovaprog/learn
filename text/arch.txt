
arch install grub (after basic installation):

mount /dev/sda1 /mnt #sda1 is `boot` partition
arch-chroot /mnt
pacman -S os-prober
grub-install /dev/sda
grub-mkconfig -o /boot/grub/grub.cfg
exit
reboot
